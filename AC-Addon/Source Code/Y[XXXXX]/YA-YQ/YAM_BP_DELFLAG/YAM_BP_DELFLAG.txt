        SET    kzloesch = abap_true

        MESSAGE `Error in update!` TYPE 'I'.
    ENDTRY.

    IF sy-subrc EQ 0.
      SELECT * FROM ihpa INTO TABLE gt_ihpa1 WHERE objnr = s_data-value
                                              AND  aedat = p_aedat.
      IF sy-subrc = 0.
        APPEND LINES OF gt_ihpa1 TO gt_ihpa.
        REFRESH gt_ihpa1.
      ENDIF.
    ENDIF.
