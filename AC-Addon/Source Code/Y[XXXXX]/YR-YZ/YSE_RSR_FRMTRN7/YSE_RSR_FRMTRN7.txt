REPORT  YSE_RSR_FRMTRN7.

TABLES : MARA.

SELECT-OPTIONS : S_MATNR FOR MARA-MATNR.


DATA : GT_MARA TYPE TABLE OF MARA.


DATA : FM_NAME TYPE RS38L_FNAM,
      FP_OUTPUTPARAMS   TYPE SFPOUTPUTPARAMS,
      FP_DOCPARAMS      TYPE SFPDOCPARAMS,
      ERROR_STRING      TYPE STRING.




START-OF-SELECTION.

  SELECT * FROM MARA INTO TABLE GT_MARA WHERE MATNR IN S_MATNR.





*   Get the name of the function module generated from the adobe form
  CALL FUNCTION 'FP_FUNCTION_MODULE_NAME'
    EXPORTING
      I_NAME     = 'YSE_RSR_FRMTRN7'
    IMPORTING
      E_FUNCNAME = FM_NAME.








* Set output parameters and open spool job
  FP_OUTPUTPARAMS-NODIALOG = 'X'.    " suppress printer dialog popup
  FP_OUTPUTPARAMS-PREVIEW  = 'X'.    " launch print preview


  CALL FUNCTION 'FP_JOB_OPEN'
    CHANGING
      IE_OUTPUTPARAMS = FP_OUTPUTPARAMS
    EXCEPTIONS
      CANCEL          = 1
      USAGE_ERROR     = 2
      SYSTEM_ERROR    = 3
      INTERNAL_ERROR  = 4
      OTHERS          = 5.



* Set form language and country (->form locale)
  FP_DOCPARAMS-LANGU   = 'E'.
  FP_DOCPARAMS-COUNTRY = 'GB'.




* Now call the generated function module

 CALL FUNCTION FM_NAME
    EXPORTING
    /1BCDWB/DOCPARAMS        =  FP_DOCPARAMS
      GT_MARA                  =  GT_MARA
* IMPORTING
*   /1BCDWB/FORMOUTPUT       =
* EXCEPTIONS
*   USAGE_ERROR              = 1
*   SYSTEM_ERROR             = 2
*   INTERNAL_ERROR           = 3
*   OTHERS                   = 4
          .
IF SY-SUBRC <> 0.
* MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
*         WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
ENDIF.






* If the call was not successful, let us get the failure info.
  IF SY-SUBRC <> 0.
    CALL FUNCTION 'FP_GET_LAST_ADS_ERRSTR'
      IMPORTING
        E_ADSERRSTR = ERROR_STRING.
    IF NOT ERROR_STRING IS INITIAL.
*     we received a detailed error description
      WRITE:/ ERROR_STRING.
      EXIT.
    ENDIF.
  ENDIF.


* Close spool job
  CALL FUNCTION 'FP_JOB_CLOSE'
    EXCEPTIONS
      USAGE_ERROR    = 1
      SYSTEM_ERROR   = 2
      INTERNAL_ERROR = 3
      OTHERS         = 4.
  IF SY-SUBRC <> 0.
    MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
            WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
  ENDIF.
