PARAMETERS p_datum TYPE i.

DATA:  g_date          LIKE sy-datum.

DELETE FROM yse_close_log WHERE datum LT g_date.

  WRITE: / text-e01.              "No entries to delete
ELSE.
  WRITE: / text-w01, g_date.      "all Entries deleted older then <g_date>
ENDIF.
