FORM UPLOAD_FILE .
  MOVE P_FILE TO S_FILE.
  CALL FUNCTION 'GUI_UPLOAD'
    EXPORTING
      FILENAME                      = S_FILE
     FILETYPE                      = 'ASC'
     HAS_FIELD_SEPARATOR           = 'X'
*   HEADER_LENGTH                 = 0
*   READ_BY_LINE                  = 'X'
*   DAT_MODE                      = ' '
*   CODEPAGE                      = ' '
*   IGNORE_CERR                   = ABAP_TRUE
*   REPLACEMENT                   = '#'
*   CHECK_BOM                     = ' '
* IMPORTING
*   FILELENGTH                    =
*   HEADER                        =
    TABLES
      DATA_TAB                      = IT_VEND1
 EXCEPTIONS
   FILE_OPEN_ERROR               = 1
   FILE_READ_ERROR               = 2
   NO_BATCH                      = 3
   GUI_REFUSE_FILETRANSFER       = 4
   INVALID_TYPE                  = 5
   NO_AUTHORITY                  = 6
   UNKNOWN_ERROR                 = 7
   BAD_DATA_FORMAT               = 8
   HEADER_NOT_ALLOWED            = 9
   SEPARATOR_NOT_ALLOWED         = 10
   HEADER_TOO_LONG               = 11
   UNKNOWN_DP_ERROR              = 12
   ACCESS_DENIED                 = 13
   DP_OUT_OF_MEMORY              = 14
   DISK_FULL                     = 15
   DP_TIMEOUT                    = 16
   OTHERS                        = 17.
  .
  IF SY-SUBRC <> 0.
* MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
*         WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
  ENDIF.



  LOOP AT IT_VEND1 INTO WA_VEND.
    IF WA_VEND-ID EQ '1'.
*Check if the vendor exists in SAP
*Use SORT2 field, this is the unique scala number
      TRANSLATE WA_VEND-SORT2 TO UPPER CASE.
      CLEAR LFA1-LIFNR.
      SELECT SINGLE LIFNR FROM LFA1 INTO LFA1-LIFNR
                  WHERE SORTL EQ WA_VEND-SORT2.
      IF SY-SUBRC EQ 0.
        WA_VEND_EXTRA-LIFNR = LFA1-LIFNR.
        APPEND WA_VEND_EXTRA TO IT_VEND_EXTRA.
      ENDIF.
    ENDIF.

    CLEAR: WA_VEND.
  ENDLOOP.

ENDFORM.                    " upload_file
*&---------------------------------------------------------------------*
*&      Form  generate_iban
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM GENERATE_IBAN .


  LOOP AT IT_VEND_EXTRA INTO WA_VEND_EXTRA.
    FREE BDCDATA.

    PERFORM BDC_DYNPRO      USING 'SAPMF02K'
                                  '0101'.
    PERFORM BDC_FIELD       USING 'BDC_OKCODE'
                                  '/00'.
    PERFORM BDC_FIELD       USING 'RF02K-LIFNR'
                                  WA_VEND_EXTRA-LIFNR.
    PERFORM BDC_FIELD       USING 'RF02K-D0130'
                                  'X'.

*------------------------------
    PERFORM BDC_DYNPRO      USING 'SAPMF02K'
                                  '0130'.
    PERFORM BDC_FIELD       USING 'BDC_OKCODE'
                                  '=IBAN'.

*------------------------------
    PERFORM BDC_DYNPRO      USING 'SAPLIBMA'
                                  '0100'.
    PERFORM BDC_FIELD       USING 'BDC_OKCODE'
                                  '=ENTR'.


*------------------------------
    PERFORM BDC_DYNPRO      USING 'SAPMF02K'
                                  '0130'.
    PERFORM BDC_FIELD       USING 'BDC_OKCODE'
                                  '=UPDA'.

    BDC_OPTIONS-DISMODE = P_MODE.

    CALL TRANSACTION 'XK02' USING BDCDATA
                            OPTIONS FROM BDC_OPTIONS
                            MESSAGES INTO IT_MESSAGES.

    CLEAR WA_VEND_EXTRA.
  ENDLOOP.

ENDFORM.                    " generate_iban




*&---------------------------------------------------------------------*
*&      Form  BDC_DYNPRO
*&---------------------------------------------------------------------*
*       Start new screen
*----------------------------------------------------------------------*
FORM BDC_DYNPRO USING PROGRAM DYNPRO.
  CLEAR BDCDATA.
  BDCDATA-PROGRAM  = PROGRAM.
  BDCDATA-DYNPRO   = DYNPRO.
  BDCDATA-DYNBEGIN = 'X'.
  APPEND BDCDATA.
ENDFORM.                    "bdc_dynpro
*&---------------------------------------------------------------------*
*&      Form  BDC_FIELD
*&---------------------------------------------------------------------*
*       Insert field
*----------------------------------------------------------------------*
FORM BDC_FIELD USING FNAM FVAL.
  IF FVAL <> NODATA_CHARACTER.
    CLEAR BDCDATA.
    BDCDATA-FNAM = FNAM.
    BDCDATA-FVAL = FVAL.
    APPEND BDCDATA.
  ENDIF.
ENDFORM.                    "bdc_field
