***********************************************************************
* INITIALIZATION                                                      *
***********************************************************************
INITIALIZATION.
  gv_repid = sy-repid.

  it_auart-auart = 'ZQP'.
  APPEND it_auart.
  it_auart-auart = 'ZQP2'.
  APPEND it_auart.
  it_auart-auart = 'ZRIB'.
  APPEND it_auart.

  SELECT tvak~auart tvakt~bezei
      INTO TABLE it_auart
      FROM tvak
      INNER JOIN tvakt
      ON tvak~auart = tvakt~auart
      FOR ALL entries IN it_auart
      WHERE tvak~auart  EQ it_auart-auart
      AND tvakt~spras EQ 'E'.

  it_pstyv-pstyv = 'ZMVN'.
  APPEND it_pstyv.
  it_pstyv-pstyv = 'ZNOD'.
  APPEND it_pstyv.
  it_pstyv-pstyv = 'ZMIL'.    "Milestone Billing With Delivery
  APPEND it_pstyv.
  it_pstyv-pstyv = 'ZMWD'.    "Milestone Billing W/O Delivery
  APPEND it_pstyv.
* begin of insert MOD-004
  it_pstyv-pstyv = 'ZRNS'.    "Rental w/o serial number
  APPEND it_pstyv.
  it_pstyv-pstyv = 'ZMNS'.    "Rental Milestone w/o D
  APPEND it_pstyv.
* end of insert MOD-004

  SELECT tvap~pstyv tvapt~vtext
      INTO TABLE it_pstyv
      FROM tvap
      INNER JOIN tvapt
      ON tvap~pstyv = tvapt~pstyv
      FOR ALL entries IN it_pstyv
      WHERE tvap~pstyv  EQ it_pstyv-pstyv
      AND tvapt~spras EQ 'E'.


***********************************************************************
* AT SELECTION SCREEN                                                 *
***********************************************************************
AT SELECTION-SCREEN.

  PERFORM check_authorisation.



AT SELECTION-SCREEN ON VALUE-REQUEST FOR so_auart-low.

  CLEAR it_return_tab.

  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield        = 'AUART'
      window_title    = 'Contract type'(029)
      value_org       = 'S'
    TABLES
      value_tab       = it_auart
      return_tab      = it_return_tab
    EXCEPTIONS
      parameter_error = 1
      no_values_found = 2
      OTHERS          = 3.

  IF sy-subrc EQ 0.
    READ TABLE it_return_tab INDEX 1.
    so_auart-low = it_return_tab-fieldval.
  ENDIF.


AT SELECTION-SCREEN ON VALUE-REQUEST FOR so_pstyv-low.

  CLEAR it_return_tab.

  CALL FUNCTION 'F4IF_INT_TABLE_VALUE_REQUEST'
    EXPORTING
      retfield        = 'PSTYV'
      window_title    = 'Contract type'(030)
      value_org       = 'S'
    TABLES
      value_tab       = it_pstyv
      return_tab      = it_return_tab
    EXCEPTIONS
      parameter_error = 1
      no_values_found = 2
      OTHERS          = 3.

  IF sy-subrc EQ 0.
    READ TABLE it_return_tab INDEX 1.
    so_pstyv-low = it_return_tab-fieldval.
  ENDIF.


***********************************************************************

***********************************************************************
START-OF-SELECTION.

* Initialize parameters
  PERFORM init_params.

* Load input data
  PERFORM load_data.

* Display results
  CALL SCREEN 100.
