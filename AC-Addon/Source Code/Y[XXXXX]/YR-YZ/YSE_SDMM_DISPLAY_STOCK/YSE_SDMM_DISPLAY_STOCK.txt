*  perform get_data.

*  gv_repid = sy-repid.
*  if p_detail = 'X'.

*  endif.

*  if p_daily = 'X' or p_sum = 'X'.
*    call screen 200.
*  endif.

  if p_detail = 'X'.
    gv_program_name = 'YSE_SDMM_DISPLAY_STOCK_DETAIL'.
  endif.

*  if p_daily = 'X'.
*    gv_program_name = 'YSE_SDMM_DISPLAY_STOCK_DAILY'.
*  endif.

  if p_sum = 'X'.
    gv_program_name = 'YSE_SDMM_DISPLAY_STOCK_SUMMARY'.
  endif.

  data: lv_number           type tbtcjob-jobcount,
        lv_print_parameter  type pri_params,
        lv_jname            type tbtcjob-jobname.

  lv_jname = gv_program_name.

  if sy-batch is not initial.
    call function 'JOB_OPEN'
      exporting
        jobname                =  lv_jname
      importing
        jobcount               =  lv_number
      exceptions
        cant_create_job        = 1
        invalid_job_data       = 2
        jobname_missing        = 3
        others                 = 4
               .
    if sy-subrc <> 0.
      message id  sy-msgid type sy-msgty number sy-msgno
              with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
    endif.

    submit (gv_program_name) with s_matnr in s_matnr
                             with s_werks in s_werks
                             with p_var = p_var
               to sap-spool
               spool parameters lv_print_parameter
               without spool dynpro
               via job  lv_jname number lv_number
               and return.
    if sy-subrc = 0.
      call function 'JOB_CLOSE'
        exporting
          jobcount                          = lv_number
          jobname                           = lv_jname
          STRTIMMED                         = 'X'
       exceptions
         cant_start_immediate              = 1
         invalid_startdate                 = 2
         jobname_missing                   = 3
         job_close_failed                  = 4
         job_nosteps                       = 5
         job_notex                         = 6
         lock_failed                       = 7
         invaild_target                    = 8
         others                            = 9
                .
      if sy-subrc <> 0.
        message id sy-msgid type sy-msgty number sy-msgno
                with sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
      endif.
    endif.
  else.
    submit (gv_program_name) with s_matnr in s_matnr
                             with s_werks in s_werks
                             with p_var = p_var
               and return.
  endif.
