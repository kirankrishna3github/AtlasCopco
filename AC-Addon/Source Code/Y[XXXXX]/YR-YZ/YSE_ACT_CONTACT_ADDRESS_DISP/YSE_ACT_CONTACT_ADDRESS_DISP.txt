      gv_ifile         LIKE /sapdmc/lsoinp-filename,

************************************************************************
*       I N I T I A L I Z A T I O N    E V E N T                       *
************************************************************************
INITIALIZATION.

* Logical system
  CALL FUNCTION 'OWN_LOGICAL_SYSTEM_GET'
    IMPORTING
      own_logical_system             = gv_logsys
    EXCEPTIONS
      own_logical_system_not_defined = 1
      OTHERS                         = 2.
  IF sy-subrc <> 0.
    MESSAGE ID sy-msgid TYPE sy-msgty NUMBER sy-msgno
         WITH sy-msgv1 sy-msgv2 sy-msgv3 sy-msgv4.
  ENDIF.


************************************************************************
*       S T A R T - O F - S E L E C T I O N    E V E N T               *
************************************************************************
START-OF-SELECTION.

* Select data
  PERFORM select_data.

* Check anything selected
  IF gt_in[] IS INITIAL.
    MESSAGE ID 'YSE_GENERAL' TYPE 'S' NUMBER '000'
            WITH 'No contacts selected'(e01).
    RETURN.
  ENDIF.

* Display output.
  PERFORM display_output.


************************************************************************
*       S U B R O U T I N E S                                          *
************************************************************************

*&---------------------------------------------------------------------*
*&      Form  SELECT_DATA
*&---------------------------------------------------------------------*
*       Select data
*----------------------------------------------------------------------*
FORM select_data .

* Input file
  PERFORM upload_data.
  CHECK gv_error IS INITIAL.
  CHECK NOT gt_in[] IS INITIAL.
  SORT gt_in.
  DELETE ADJACENT DUPLICATES FROM gt_in.

ENDFORM.                    " SELECT_DATA

*&---------------------------------------------------------------------*
*&      Form  UPLOAD_DATA
*&---------------------------------------------------------------------*
*       Upload iUNIX file
*----------------------------------------------------------------------*
FORM upload_data .

  CONCATENATE 'CONTACT' 'ADDRESS' p_vkorg
              INTO gv_ifile SEPARATED BY '_'.

  REPLACE 'xxx' IN gv_directory WITH gv_logsys(3).
  CONCATENATE gv_directory gv_ifile INTO gv_ifile.

  OPEN DATASET gv_ifile FOR INPUT IN TEXT MODE ENCODING DEFAULT.
  IF sy-subrc <> 0.
    WRITE: / text-e02, gv_ifile.
    gv_error = 'X'.
    EXIT.
  ENDIF.

  DO.
    READ DATASET gv_ifile INTO gt_in.
    IF sy-subrc NE 0.
      EXIT.
    ENDIF.

    APPEND gt_in.
  ENDDO.


ENDFORM.                    " UPLOAD_DATA

*&---------------------------------------------------------------------*
*&      Form  DISPLAY_OUTPUT
*&---------------------------------------------------------------------*
*       Display input file
*----------------------------------------------------------------------*
FORM display_output .

  LOOP AT gt_in.
    CHECK gt_in-vkorg = p_vkorg.
    IF gt_in-country_p IS INITIAL.
      gt_in-country_p = 'P C'.
    ENDIF.
    WRITE: /01  gt_in-vkorg,
                gt_in-kunnr,
                gt_in-sortl,
                gt_in-ktokd,
                gt_in-name1,
                gt_in-name2,
                gt_in-parnr,
                gt_in-namel,
                gt_in-namev,
                gt_in-tel_number,
                gt_in-tel_extens,
                gt_in-mob_number,     "Mobile
                gt_in-anred,
                gt_in-title_aca1,
                gt_in-parla,
                gt_in-parge,
                gt_in-parvo,
                gt_in-pavip,
                gt_in-titel_ap,
                gt_in-abtnr,
                gt_in-abtpa,
                gt_in-gbdat,
                gt_in-fax_number,
                gt_in-smtp_addr,
                gt_in-deflt_comm,
                gt_in-street_b,     "Business
                gt_in-str_sup_b,
                gt_in-city1_b,
                gt_in-country_b,
                gt_in-street_p,     "Private
                gt_in-post_cd_p,
                gt_in-city1_p,
                gt_in-country_p.

  ENDLOOP.

ENDFORM.                    " SAVE_OUTPUT
