*&---------------------------------------------------------------------*
*&      Form  get_acc_ind
*&---------------------------------------------------------------------*
*       Get accounting indicator
*----------------------------------------------------------------------*
FORM get_acc_ind .

  CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
    EXPORTING
      input  = gt_irec-aufnr
    IMPORTING
      output = gv_aufnr.

  SELECT SINGLE bemot INTO gv_bemot
          FROM aufk
          WHERE aufnr = gv_aufnr.

ENDFORM.                    " get_acc_ind

*&---------------------------------------------------------------------*
*&      Form  get_stor_loc
*&---------------------------------------------------------------------*
*       Get storage location
*----------------------------------------------------------------------*
FORM get_stor_loc.

  CALL FUNCTION 'CONVERSION_EXIT_ALPHA_INPUT'
    EXPORTING
      input  = gt_irec-matnr
    IMPORTING
      output = gv_matnr.

  SELECT single lgort INTO gv_lgort
          FROM resb
          WHERE aufnr = gv_aufnr
            and matnr = gv_matnr
            and werks = p_werks
            and xloek = space
            and vornr = c_vornr.

ENDFORM.                    " get_stor_loc
