REPORT  yrd_pp_kit_label_sf.

*&---------------------------------------------------------------------*
*& DEFINE TABLES.                                                      *
*&---------------------------------------------------------------------*
TABLES: afko.

DATA: it_kit TYPE TABLE OF yrd_pp_kit_lable WITH HEADER LINE.
DATA: it_kit_prt TYPE TABLE OF yrd_pp_kit_lable WITH HEADER LINE.
CONSTANTS: c_tdsfname TYPE tdsfname VALUE 'YRD_PP_KIT_LABEL'.



*&---------------------------------------------------------------------*
*&      DEFINE PARAMETERS                                              *
*&---------------------------------------------------------------------*
*Plant
PARAMETERS p_werks TYPE werks OBLIGATORY VALUE CHECK DEFAULT 'CN61'.
*Print Copies
PARAMETERS p_copies TYPE i.
*Production Order
SELECT-OPTIONS s_aufnr FOR afko-aufnr.



*&---------------------------------------------------------------------*
*& INITIALIZATION                                                      *
*&---------------------------------------------------------------------*
INITIALIZATION.
*  PERFORM initialization.
*&---------------------------------------------------------------------*
*& AT SELECTION-SCREEN                                                 *
*&---------------------------------------------------------------------*
AT SELECTION-SCREEN.
*  PERFORM check_input.
*&---------------------------------------------------------------------*
*& START-OF-SELECTION                                                  *
*&---------------------------------------------------------------------*
START-OF-SELECTION.
  PERFORM get_data.
  PERFORM sf_output.



*&---------------------------------------------------------------------*
*&      Form  GET_DATA
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM get_data .

  CLEAR: it_kit,it_kit[],it_kit_prt,it_kit_prt[].

  SELECT *
    FROM afpo
    INTO CORRESPONDING FIELDS OF TABLE it_kit
   WHERE pwerk = p_werks
     AND aufnr IN s_aufnr.

  LOOP AT it_kit.
    IF p_copies IS INITIAL.
      DO it_kit-psmng TIMES.
        MOVE it_kit TO it_kit_prt.
        APPEND it_kit_prt.
        CLEAR it_kit_prt.
      ENDDO.
    ELSE.
      DO p_copies TIMES.
        MOVE it_kit TO it_kit_prt.
        APPEND it_kit_prt.
        CLEAR it_kit_prt.
      ENDDO.
    ENDIF.
  ENDLOOP.

ENDFORM.                    " GET_DATA
*&---------------------------------------------------------------------*
*&      Form  SF_OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
*  -->  p1        text
*  <--  p2        text
*----------------------------------------------------------------------*
FORM sf_output .

  DATA: l_fname TYPE rs38l_fnam.

  CALL FUNCTION 'SSF_FUNCTION_MODULE_NAME'
    EXPORTING
      formname                 = c_tdsfname
*     VARIANT                  = ' '
*     DIRECT_CALL              = ' '
    IMPORTING
      fm_name                  = l_fname
    EXCEPTIONS
      no_form                  = 1
      no_function_module       = 2
      OTHERS                   = 3.
  IF sy-subrc <> 0.
* MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
* WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
  ENDIF.

  CALL FUNCTION l_fname
    TABLES
      it_kit           = it_kit_prt
    EXCEPTIONS
      formatting_error = 1
      internal_error   = 2
      send_error       = 3
      user_canceled    = 4
      OTHERS           = 5.
  IF sy-subrc <> 0.
* MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
* WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
  ENDIF.


ENDFORM.                    " SF_OUTPUT
