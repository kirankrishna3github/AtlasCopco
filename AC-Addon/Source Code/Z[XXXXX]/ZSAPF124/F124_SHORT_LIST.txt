Form Sum_info using I_bukrs like T001-Bukrs
                    I_level type c
                    E_sum_op type i
                    E_sum_ap type i
                    E_sum_eap type i.
   Check Flg_liste ne char_3.

   If Xtol = 'X' and I_level = '1'.
      E_sum_ap = E_sum_eap.
      If X_echtl ne 'X'.
         Clear E_sum_eap.
      Endif.
   Endif.
   Percent = ( 1000 * E_sum_ap ) / E_sum_op.
   If X_echtl = 'X' and Enqsubrc = 0.
      Spercent = ( 1000 * E_sum_eap ) / E_sum_op.
   Endif.
   Write:/01(01) sy-vline.
   Case I_level.
   When Char_0.
      Format color col_total intensified.     "Summe
      Write: 02(05) char_*,
             07(01) sy-vline,
             08(05) char_*.
   When Char_1.
      Format color col_total intensified off. "Zwischensumme
      Write: 02(05) I_bukrs,
             07(01) sy-vline,
             08(05) Koart.
   Endcase.
   Write:    13(01) sy-vline,
             14(10) char_*,
             24(01) sy-vline,
             25(10) char_*,
             35(01) sy-vline,
             36(12) E_sum_op,
             48(01) sy-vline,
             49(12) E_sum_ap,
             61(01) sy-vline,
             62(12) Percent,
             74(01) sy-vline,
             75(19) E_sum_eap,
             94(01) sy-vline,
             95(13) Spercent,
            108(01) sy-vline,
            153(01) sy-vline.
   Case I_level.
   When char_0.
      Uline.
   Endcase.
   Case Koart.
   When 'D'.
      Sumd_op = Sumd_op + E_sum_op.
      Sumd_ap = Sumd_ap + E_sum_ap.
      Sumd_eap = Sumd_eap + E_sum_eap.
   When 'K'.
      Sumk_op = Sumk_op + E_sum_op.
      Sumk_ap = Sumk_ap + E_sum_ap.
      Sumk_eap = Sumk_eap + E_sum_eap.
   When 'S'.
      Sums_op = Sums_op + E_sum_op.
      Sums_ap = Sums_ap + E_sum_ap.
      Sums_eap = Sums_eap + E_sum_eap.
   Endcase.
   Clear E_sum_op.
   Clear E_sum_ap.
   Clear E_sum_eap.
Endform.                               "SUM_INFO
