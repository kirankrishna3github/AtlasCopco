REPORT  zzreset_rel                             .

DATA: gt_status TYPE STANDARD TABLE OF jstat
        INITIAL SIZE 0 WITH HEADER LINE.

DATA: gv_objnr TYPE caufv-objnr.

PARAMETERS: p_aufnr TYPE caufv-aufnr,
            p_reset TYPE c.

SELECT SINGLE objnr
INTO gv_objnr
FROM caufv
WHERE aufnr EQ p_aufnr.

IF p_reset EQ 'X'.
  gt_status-stat = 'I0001'.
  gt_status-inact = space.
  APPEND gt_status.
  gt_status-stat = 'I0002'.
  gt_status-inact = 'X'.
  APPEND gt_status.
  update afih set iphas = '0'
  where aufnr eq p_aufnr.
ELSE.
  gt_status-stat = 'I0001'.
  gt_status-inact = 'X'.
  APPEND gt_status.
  gt_status-stat = 'I0002'.
  gt_status-inact = space.
  APPEND gt_status.
  update afih set iphas = '2'
  where aufnr eq p_aufnr.
ENDIF.

CALL FUNCTION 'STATUS_CHANGE_INTERN'
  EXPORTING
*   CHECK_ONLY                = ' '
*   CLIENT                    = SY-MANDT
    objnr                     = gv_objnr
*   ZEILE                     = ' '
*   SET_CHGKZ                 =
* IMPORTING
*   ERROR_OCCURRED            =
*   OBJECT_NOT_FOUND          =
*   STATUS_INCONSISTENT       =
*   STATUS_NOT_ALLOWED        =
  TABLES
    status                    = gt_status
  EXCEPTIONS
    object_not_found          = 1
    status_inconsistent       = 2
    status_not_allowed        = 3
    OTHERS                    = 4
          .
IF sy-subrc <> 0.
  MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
          WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
ENDIF.

CALL FUNCTION 'STATUS_UPDATE_ON_COMMIT'
  .

COMMIT WORK.
